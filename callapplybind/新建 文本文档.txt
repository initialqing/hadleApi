版本一：实现了改变this指向的功能
Function.prototype.MyBind = function (obj) {
    const self = this
    return function () {
        self.call(obj)
    }
}

function testBind(a = 1, b = 2) {
    console.log('this:', this.name);
    return a + b
}

const bindObj = { name: 'qing', age: 18 };
const bindFun = testBind.MyBind(bindObj)
bindFun()
